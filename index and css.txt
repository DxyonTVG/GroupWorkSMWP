<script>
    function handleText(field, messageDisplay) {
        const valueInserted = document.getElementById(field).value;

      
        removeMessage(messageDisplay);

       
        if (field === 'email') {
            if (!valueInserted) {
                document.getElementById(messageDisplay).innerText = "Email is required.";
            }   (!valueInserted.includes('@') || !valueInserted.endsWith('.com'))
                document.getElementById(messageDisplay).innerText = "Email must be in the format letters@something.com.";

        } else if (field === 'name') {
            if (!valueInserted) {
                document.getElementById(messageDisplay).innerText = "Name is required.";
            } else if (!/^[a-zA-Z\s]+$/.test(valueInserted)) {
                document.getElementById(messageDisplay).innerText = "Name can only contain letters.";
            }
        } else if (field === 'phone') {
            if (valueInserted && !/^\d+$/.test(valueInserted)) {
                document.getElementById(messageDisplay).innerText = "Phone can only be numbers.";
            }
        } else if (field === 'age' && !valueInserted) {
            document.getElementById(messageDisplay).innerText = "Age is required.";
        } else if (field === 'address') {
            if (!valueInserted) {
                document.getElementById(messageDisplay).innerText = "Address is required.";
            } else if (/^\d+$/.test(valueInserted)) {
                document.getElementById(messageDisplay).innerText = "Address cannot be just numbers.";
            }
        }
    }
    function removeMessage(messageDisplay) {
        document.getElementById(messageDisplay).innerText = "";
    }
    <!-- Lanyah M "added JavaScript, and included some regex to handle what can be inputted by a user" -->
</script>
